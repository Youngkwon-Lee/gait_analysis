# 모델 파일 문제 해결

## 문제
[ERROR] Model not found: /home2/gun3856/gait_code/models/OA_Screening_best.pth

---

## 해결 방법

### 1단계: 모델 파일 확인
ls -lh ~/gait_code/models/

---

### 2-A: 모델이 있는 경우 (다른 이름으로 저장됨)
# 파일 이름 확인 후 코드 수정 또는 심볼릭 링크 생성
ln -s ~/gait_code/models/실제파일명.pth ~/gait_code/models/OA_Screening_best.pth

---

### 2-B: 모델이 없는 경우 - Baseline 학습 먼저 실행

cd ~/gait_code
nohup python -u src/train_baseline_hpc.py --task OA_Screening > logs/train_oa.log 2>&1 &

# 학습 진행 확인 (1-2분 후)
tail -f logs/train_oa.log

# 학습 완료 확인 (30-60분 후)
tail -100 logs/train_oa.log | grep "Best"

# 모델 파일 생성 확인
ls -lh ~/gait_code/models/

---

### 3단계: 학습 완료 후 다시 Error Analysis 실행

cd ~/gait_code && export DATA_PATH=~/gait_code/dataset/data && export OUTPUT_PATH=~/gait_code/results/error_analysis && export MODEL_PATH=~/gait_code/models && nohup python -u src/analyze_errors.py > logs/error_analysis.log 2>&1 &

cd ~/gait_code && export DATA_PATH=~/gait_code/dataset/data && export OUTPUT_PATH=~/gait_code/results/confusion_analysis && export MODEL_PATH=~/gait_code/models && nohup python -u src/analyze_confusion.py > logs/confusion_analysis.log 2>&1 &

---
